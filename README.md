var DIMENSION = 2;
var i;

function get( i,  j) {
	var N = DIMENSION;
	var index = i * N + j;
	var value = Response[index];
	return value;
}
 
function distance(  v1,   v2) {
	var sum = 0.0;
	for (var i = 0; i < DIMENSION; i++) {
		var x1 = get(v1, i);
		var x2 = get(v2, i);
		var dx = x1 - x2;
		sum += dx * dx;
	}
	return Math.sqrt(sum);
}

}
